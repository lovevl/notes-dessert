对象原型属性屏蔽

  function Product(){

        this.name='神仙';

        this.description = '';

        this.buy = function(){

            alert('构造函数对象')

        }

    }

    Product.prototype={

        name:'王书奎',

        buy:function(){

            alert('原型对象')

        }

    }

    var product = new Product()

    /*原型属性屏蔽理论 -- 乌云蔽日*/

    console.log(product.buy())    // 构造函数对象

    /*清除乌云*/

    delete product.buy;

    console.log(product.buy())     //原型对象

    /*被屏蔽之后如何获取 */

    console.log(Product.prototype.buy())   //原型对象
